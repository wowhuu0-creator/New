import { useState, useRef, useEffect } from "react";

const API_URL = "https://api.anthropic.com/v1/messages";

async function callClaude(question, mode) {
  let systemPrompt, userPrompt;

  if (mode === "answer") {
    systemPrompt =
      "You are a helpful, knowledgeable assistant. Give clear, thorough answers in 2-4 paragraphs. Be informative and educational.";
    userPrompt = question;
  } else if (mode === "mindmap") {
    systemPrompt = `You generate mind map data as JSON ONLY. No extra text, no markdown fences.
Return this exact structure:
{"center":"Topic Title","branches":[{"label":"Branch 1","leaves":["detail a","detail b"]},{"label":"Branch 2","leaves":["detail c","detail d"]},{"label":"Branch 3","leaves":["detail e","detail f"]},{"label":"Branch 4","leaves":["detail g","detail h"]}]}
Max 5 branches, 3 leaves each. Keep labels short (2-5 words).`;
    userPrompt = `Create a mind map for: ${question}`;
  } else {
    systemPrompt = `You generate structured notes as JSON ONLY. No extra text, no markdown fences.
Return this exact structure:
{"cards":[{"type":"summary","title":"Overview","points":["point 1","point 2","point 3"]},{"type":"concept","title":"Key Concept","points":["detail 1","detail 2"]},{"type":"keypoint","title":"Important Points","points":["key 1","key 2","key 3"]}]}
Max 4 cards. Keep points concise.`;
    userPrompt = `Create structured notes for: ${question}`;
  }

  const res = await fetch(API_URL, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      system: systemPrompt,
      messages: [{ role: "user", content: userPrompt }],
    }),
  });

  if (!res.ok) {
    const err = await res.text();
    throw new Error(`API ${res.status}: ${err}`);
  }
  const data = await res.json();
  const text = data.content.map((c) => c.text || "").join("").trim();

  if (mode === "mindmap" || mode === "notes") {
    const clean = text.replace(/```json|```/g, "").trim();
    return JSON.parse(clean);
  }
  return text;
}

// ── Mindmap SVG renderer ──
function MindmapSVG({ data }) {
  const svgRef = useRef(null);
  const [dims, setDims] = useState({ w: 600, h: 480 });

  useEffect(() => {
    if (svgRef.current) {
      const { width, height } = svgRef.current.getBoundingClientRect();
      setDims({ w: width || 600, h: height || 480 });
    }
  }, [data]);

  if (!data) return null;

  const { w, h } = dims;
  const cx = w / 2, cy = h / 2;
  const branches = data.branches || [];
  const angleStep = (2 * Math.PI) / Math.max(branches.length, 1);
  const branchR = Math.min(w, h) * 0.28;
  const leafR = Math.min(w, h) * 0.16;

  const nodes = [];
  const links = [];

  // Central node
  nodes.push({ x: cx, y: cy, label: data.center, type: "central" });

  branches.forEach((branch, i) => {
    const angle = i * angleStep - Math.PI / 2;
    const bx = cx + branchR * Math.cos(angle);
    const by = cy + branchR * Math.sin(angle);

    links.push({ x1: cx, y1: cy, x2: bx, y2: by, isLeaf: false });
    nodes.push({ x: bx, y: by, label: branch.label, type: "branch" });

    (branch.leaves || []).forEach((leaf, j) => {
      const leafCount = branch.leaves.length;
      const spread = Math.PI / 5;
      const leafAngle = angle + (j - (leafCount - 1) / 2) * spread;
      const lx = bx + leafR * Math.cos(leafAngle);
      const ly = by + leafR * Math.sin(leafAngle);
      links.push({ x1: bx, y1: by, x2: lx, y2: ly, isLeaf: true });
      nodes.push({ x: lx, y: ly, label: leaf, type: "leaf" });
    });
  });

  function wrapText(label, maxChars) {
    const words = label.split(" ");
    const lines = [];
    let line = "";
    words.forEach((word) => {
      if ((line + " " + word).trim().length > maxChars) {
        if (line) lines.push(line);
        line = word;
      } else {
        line = (line + " " + word).trim();
      }
    });
    if (line) lines.push(line);
    return lines;
  }

  function curvedPath(x1, y1, x2, y2) {
    const mx = (x1 + x2) / 2, my = (y1 + y2) / 2;
    const dx = x2 - x1, dy = y2 - y1;
    const cpx = mx - dy * 0.15, cpy = my + dx * 0.15;
    return `M${x1},${y1} Q${cpx},${cpy} ${x2},${y2}`;
  }

  const nodeStyles = {
    central: { fill: "url(#cg)", stroke: "none", rx: 14, fontSz: 13, fw: "bold", maxChars: 12, pw: 90, ph: 38, textFill: "#fff" },
    branch:  { fill: "#1a1a2e", stroke: "#7c5cfc", rx: 9, fontSz: 11, fw: "600", maxChars: 10, pw: 76, ph: 30, textFill: "#e8e8f0" },
    leaf:    { fill: "rgba(0,229,195,0.08)", stroke: "rgba(0,229,195,0.3)", rx: 7, fontSz: 10, fw: "400", maxChars: 12, pw: 68, ph: 24, textFill: "rgba(232,232,240,0.75)" },
  };

  return (
    <svg ref={svgRef} width="100%" height="100%" style={{ display: "block" }}>
      <defs>
        <linearGradient id="cg" x1="0%" y1="0%" x2="100%" y2="100%">
          <stop offset="0%" stopColor="#7c5cfc" />
          <stop offset="100%" stopColor="#00e5c3" />
        </linearGradient>
      </defs>

      {links.map((l, i) => (
        <path
          key={i}
          d={curvedPath(l.x1, l.y1, l.x2, l.y2)}
          stroke={l.isLeaf ? "rgba(0,229,195,0.25)" : "rgba(124,92,252,0.4)"}
          strokeWidth={l.isLeaf ? 1 : 1.8}
          fill="none"
        />
      ))}

      {nodes.map((n, i) => {
        const s = nodeStyles[n.type];
        const lines = wrapText(n.label, s.maxChars);
        const lineH = s.fontSz + 4;
        const bh = Math.max(s.ph, lines.length * lineH + 14);
        const bw = Math.max(s.pw, Math.max(...lines.map((l) => l.length)) * (s.fontSz * 0.62) + 20);
        return (
          <g key={i} transform={`translate(${n.x},${n.y})`}>
            <rect
              x={-bw / 2} y={-bh / 2}
              width={bw} height={bh}
              rx={s.rx}
              fill={s.fill}
              stroke={s.stroke || "none"}
              strokeWidth={s.stroke ? 1.5 : 0}
            />
            {lines.map((line, li) => (
              <text
                key={li}
                x={0}
                y={(li - (lines.length - 1) / 2) * lineH}
                textAnchor="middle"
                dominantBaseline="middle"
                fontSize={s.fontSz}
                fontWeight={s.fw}
                fontFamily="'DM Sans', sans-serif"
                fill={s.textFill}
              >
                {line}
              </text>
            ))}
          </g>
        );
      })}
    </svg>
  );
}

// ── Note Card ──
function NoteCard({ card }) {
  const [copied, setCopied] = useState(false);
  const colors = {
    summary: { bar: "#00e5c3", tag: "rgba(0,229,195,0.15)", tagText: "#00e5c3" },
    concept: { bar: "#7c5cfc", tag: "rgba(124,92,252,0.2)", tagText: "#a78bfa" },
    keypoint: { bar: "#ff6b6b", tag: "rgba(255,107,107,0.15)", tagText: "#ff6b6b" },
  };
  const c = colors[card.type] || colors.concept;

  function copy() {
    const text = `${card.title}\n${(card.points || []).map((p) => `• ${p}`).join("\n")}`;
    navigator.clipboard.writeText(text).then(() => {
      setCopied(true);
      setTimeout(() => setCopied(false), 1500);
    });
  }

  return (
    <div style={{
      background: "#1a1a26", border: "1px solid rgba(255,255,255,0.07)",
      borderRadius: 12, padding: 16, position: "relative", overflow: "hidden",
      animation: "fadeUp 0.3s ease",
    }}>
      <div style={{
        position: "absolute", left: 0, top: 0, bottom: 0,
        width: 3, background: c.bar, borderRadius: "3px 0 0 3px",
      }} />
      <div style={{ display: "flex", alignItems: "center", gap: 8, marginBottom: 10 }}>
        <span style={{
          fontFamily: "'Syne', sans-serif", fontWeight: 700, fontSize: 13,
          textTransform: "uppercase", letterSpacing: "0.05em", color: "#e8e8f0",
        }}>
          {card.title}
        </span>
        <span style={{
          fontSize: 11, padding: "2px 8px", borderRadius: 20,
          background: c.tag, color: c.tagText, fontWeight: 500,
        }}>
          {card.type}
        </span>
      </div>
      <ul style={{ paddingLeft: 18, color: "rgba(232,232,240,0.8)", fontSize: 14, lineHeight: 1.7 }}>
        {(card.points || []).map((p, i) => <li key={i}>{p}</li>)}
      </ul>
      <button onClick={copy} style={{
        marginTop: 10, padding: "5px 12px", borderRadius: 8,
        border: "1px solid rgba(255,255,255,0.1)", background: "#12121a",
        color: "rgba(232,232,240,0.5)", fontSize: 12, cursor: "pointer",
        fontFamily: "'DM Sans', sans-serif", transition: "all 0.2s",
      }}>
        {copied ? "✓ Copied!" : "Copy Note"}
      </button>
    </div>
  );
}

  const [loading, setLoading] = useState(false);
  const [view, setView] = useState("mindmap"); // "mindmap" | "notes"
  const [mindmapData, setMindmapData] = useState(null);
  const [notesData, setNotesData] = useState(null);

      // 1. Answer
      const answer = await callClaude(q, "answer");
      setMessages((m) => [...m, { role: "ai", text: answer }]);
      setLoading(false);

      // 2. Mindmap
      setMmLoading(true);

      const mm = await callClaude(q, "mindmap");

      setMmLoading(false);

      const notes = await callClaude(q, "notes");
      setNotesData(notes);
      setNotesLoading(false);

      borderRadius: 20, padding: "5px 12px", fontSize: 12, color: "rgba(232,232,240,0.5)",
    },
    main: {
      position: "relative", zIndex: 1, flex: 1,
      display: "grid", gridTemplateColumns: "1fr 1px 1fr", overflow: "hidden",

                    <span style={{ fontFamily:"'Syne',sans-serif", fontSize:16 }}>Notes will appear here</span>
                    <span style={{ fontSize:13 }}>Ask any question and structured notes will be auto-generated.</span>
                  </div>
                )}
                {notesLoading && (
                  <div style={{ display:"flex", alignItems:"center", justifyContent:"center", height:"100%", color:"#7c5cfc", gap:8, fontSize:14 }}>
                    <span style={{ animation:"spin 1s linear infinite", display:"inline-block" }}>⟳</span> Generating notes…
                  </div>
                )}
                {notesData && !notesLoading && (
                  <div style={{ display:"flex", flexDirection:"column", gap:14 }}>
                    {(notesData.cards || []).map((card, i) => <NoteCard key={i} card={card} />)}
                  </div>
                )}
              </div>
            )}
          </div>
        </div>
      </div>
    </div>
  );
}
